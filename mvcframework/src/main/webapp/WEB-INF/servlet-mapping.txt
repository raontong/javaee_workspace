# 자바의 컬렉션 프레임웍 객체 중 Map 를 상속받은 Properties 라는 객체는 Properties
# 어떤 데이터가 = 을 기준으로 key=value의 망으로 구성될 경우, 
   좌측은 key 우측은 value 로 해석한다
 
/movie.do=com.ch.mvcframework.controller.MovieController
/food.do=com.ch.mvcframework.controller.FoodController
 
# 게시판 컨트롤러 매핑
/board/regist.do=com.ch.mvcframework.controller.RegistController
/board/list.do=com.ch.mvcframework.controller.ListController
/board/detail.do=com.ch.mvcframework.controller.DetailController
/board/delete.do=com.ch.mvcframework.controller.DeleteController
/board/edit.do=com.ch.mvcframework.controller.UpdateController

# 게시판 뷰페이지 매핑
# 자바 코드안에 뷰페이지명을 직접적으로 명시하게되면, 외부 파일의 명칭이 달라질 경우, 자바 코드가 영향을 받게됨
# 따라서 유지보수성을 위해서는 자바 클래스 안에, 파일명을 직접 등록해서는 안됨(하드 코딩 금지)

# 글등록 요청 처리시으 뷰매핑
/board/regist/result=/board/list.do

# 목록보기 요청 처리시의 뷰매핑 
/board/list/result=/board/list.jsp

# 상세보기 요청 처리시의 뷰매핑
/board/detail/result=/board/detail.jsp

# 삭제 요청 처리시의 뷰매핑 (.do 하는 이유? .do 거처야 컨트롤러를 만나고 그래야 새로 경신된 데이터를 보니까)
/board/delete/result=/board/list.do

# 수정 요청 처리시의 뷰매핑
/board/update/result=/board/detail.do



# 사원관련 컨트롤러 매핑
/emp/regist.do=com.ch.mvcframework.emp.controller.RegistController

# 사원등록 처리시의 뷰 매핑
/emp/regist/result=/emp/main.jsp
/emp/error=/emp/error.jsp
